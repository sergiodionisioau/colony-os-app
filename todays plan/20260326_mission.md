7. Graph Definition (Deterministic Flow)
# graphs/main_graph.py

from langgraph.graph import StateGraph, END
from orchestrator.llm import get_llm
from graphs.state import AgentState
from orchestrator.events import emit

llm = get_llm()

def planner(state: AgentState):
    response = llm.invoke(f"Break this task into steps: {state['input']}")
    state["plan"] = response.content.split("\n")
    emit("plan.created", state)
    return state

def executor(state: AgentState):
    results = []
    for step in state["plan"]:
        res = llm.invoke(f"Execute: {step}")
        results.append(res.content)

    state["result"] = "\n".join(results)
    emit("task.completed", state)
    return state

def build_graph():
    graph = StateGraph(AgentState)

    graph.add_node("plan", planner)
    graph.add_node("execute", executor)

    graph.set_entry_point("plan")
    graph.add_edge("plan", "execute")
    graph.add_edge("execute", END)

    return graph.compile()
8. Orchestrator Adapter (Bridge to Kernel)

This is where your system connects.

# orchestrator/runner.py

from graphs.main_graph import build_graph
from orchestrator.events import listen

graph = build_graph()

def run():
    while True:
        events = listen()

        for stream, msgs in events:
            for msg_id, data in msgs:
                if data["type"] == "task.created":
                    payload = data["data"]

                    state = {
                        "task_id": payload["task_id"],
                        "input": payload["input"],
                        "plan": [],
                        "result": "",
                        "status": "running"
                    }

                    graph.invoke(state)
