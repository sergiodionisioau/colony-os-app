Non-negotiable rules

# tools/schemas.py
from pydantic import BaseModel, HttpUrl, Field
from typing import Optional, Literal, Dict, Any, List

class ToolRequest(BaseModel):
    task_id: str
    action_id: str

    finished_at: str
    duration_ms: int
    artifact_paths: List[str] = []

11. Tool Router
# tools/router.py
from datetime import datetime, timezone
from tools.registry import TOOL_REGISTRY
from tools.schemas import ToolRequest, ToolResponse
from tools.policies import evaluate_policy
from tools.receipts import write_receipt

def run_tool(req: ToolRequest) -> ToolResponse:
    started = datetime.now(timezone.utc)

    decision = evaluate_policy(req.tool_name, req.arguments, req.dry_run)
    if decision != "allow":
        resp = ToolResponse(
            task_id=req.task_id,
            action_id=req.action_id,
            tool_name=req.tool_name,
            status="blocked",
            result={},
            error=f"Policy decision: {decision}",
            started_at=started.isoformat(),
            finished_at=datetime.now(timezone.utc).isoformat(),
            duration_ms=0,
            artifact_paths=[],
            policy_decision=decision,
        )
        write_receipt(resp)
        return resp

    tool_fn = TOOL_REGISTRY[req.tool_name]
    try:
        result = tool_fn(**req.arguments)
        finished = datetime.now(timezone.utc)
        resp = ToolResponse(
            task_id=req.task_id,
            action_id=req.action_id,
            tool_name=req.tool_name,
            status="success",
            result=result,
            error=None,
            started_at=started.isoformat(),
            finished_at=finished.isoformat(),
            duration_ms=int((finished - started).total_seconds() * 1000),
            artifact_paths=result.get("artifact_paths", []),
            policy_decision="allow",
        )
    except Exception as e:
        finished = datetime.now(timezone.utc)
        resp = ToolResponse(
            task_id=req.task_id,
            action_id=req.action_id,
            tool_name=req.tool_name,
            status="error",
            result={},
            error=str(e),
            started_at=started.isoformat(),
            finished_at=finished.isoformat(),
            duration_ms=int((finished - started).total_seconds() * 1000),
            artifact_paths=[],
            policy_decision="allow",
        )

    write_receipt(resp)
    return resp
12. Playwright Client

Use async Playwright in a managed session.

# tools/browser/playwright_client.py
from playwright.async_api import async_playwright

class PlaywrightSession:
    def __init__(self):
        self.playwright = None
        self.browser = None
        self.context = None
        self.page = None

    async def start(self, headless=True):
        self.playwright = await async_playwright().start()
        self.browser = await self.playwright.chromium.launch(headless=headless)
        self.context = await self.browser.new_context()
        self.page = await self.context.new_page()
        return self

    async def close(self):
        if self.context:
            await self.context.close()
        if self.browser:
            await self.browser.close()
        if self.playwright:
            await self.playwright.stop()

Playwright’s Python docs explicitly support launching and controlling browsers from Python, locally and in CI, with sync and async APIs.

# tools/browser/browser_tools.py
import asyncio
from pathlib import Path
from tools.browser.playwright_client import PlaywrightSession

ARTIFACT_DIR = Path("/home/coe/Agent/artifacts/browser")

async def _browser_goto_impl(url: str, wait_until: str = "load"):
    session = await PlaywrightSession().start(headless=True)
    try:
        await session.page.goto(url, wait_until=wait_until)
        title = await session.page.title()
        return {"url": session.page.url, "title": title, "artifact_paths": []}
    finally:
        await session.close()

def browser_goto(url: str, wait_until: str = "load"):

async def _browser_extract_text_impl(url: str):
    session = await PlaywrightSession().start(headless=True)
    try:
        await session.page.goto(url, wait_until="load")
        text = await session.page.locator("body").inner_text()
        return {"text": text[:50000], "artifact_paths": []}
    finally:
        await session.close()

def browser_extract_text(url: str):
    return asyncio.run(_browser_extract_text_impl(url))

async def _browser_screenshot_impl(url: str, filename: str):
    session = await PlaywrightSession().start(headless=True)
    try:
        await session.page.goto(url, wait_until="load")
        ARTIFACT_DIR.mkdir(parents=True, exist_ok=True)
        path = ARTIFACT_DIR / filename
        await session.page.screenshot(path=str(path), full_page=True)
        return {"path": str(path), "artifact_paths": [str(path)]}
    finally:
        await session.close()

def browser_screenshot(url: str, filename: str):
    return asyncio.run(_browser_screenshot_impl(url, filename))

import psycopg2

def db_query_readonly(sql: str):
    if not sql.strip().lower().startswith("select"):
        raise ValueError("Only SELECT queries are allowed")

    conn = psycopg2.connect("postgresql://user:pass@localhost:5432/coe")
    try:
        with conn, conn.cursor() as cur:
            cur.execute(sql)
            rows = cur.fetchall()
            cols = [d[0] for d in cur.description]
            return {
                "columns": cols,
                "rows": rows[:1000],
                "row_count": len(rows),
                "artifact_paths": [],
            }
    finally:
        conn.close()

16. Safe Shell Tool

This must be heavily constrained.

# tools/shell/shell_tools.py
import subprocess
import shlex

ALLOWED_BINARIES = {"ls", "cat", "grep", "find", "python3"}

def shell_run_safe(command: str, cwd: str = "/home/coe/Agent"):
    parts = shlex.split(command)
    if not parts:
        raise ValueError("Empty command")
    if parts[0] not in ALLOWED_BINARIES:
        raise ValueError(f"Binary not allowed: {parts[0]}")

    proc = subprocess.run(
        parts,
        cwd=cwd,
        capture_output=True,
        text=True,
        timeout=20,
        check=False,
    )
    return {
        "returncode": proc.returncode,
        "stdout": proc.stdout[:20000],
        "stderr": proc.stderr[:20000],
        "artifact_paths": [],
    }

Baseline rule: no shells that can mutate the machine beyond approved artifact paths.

        task_id=state["task_id"],

Edge case 1 — page never loads

Browser tool must timeout and return structured error, not hang the graph.

Edge case 2 — shell output too large

Truncate stdout/stderr and record truncation in result metadata.

If the same task_id + action_id is replayed, either:

return cached receipt, or
execute only if marked replayable.
